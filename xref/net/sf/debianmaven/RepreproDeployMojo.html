4   import java.io.IOException;
5   import java.util.ArrayList;
6   import java.util.List;

8   import org.apache.maven.plugin.MojoExecutionException;

10  import edu.emory.mathcs.backport.java.util.Arrays;
11  
12  /**
13   * @goal deploy
14   * @phase install
15   */
16  public class RepreproDeployMojo extends AbstractRepreproMojo
17  {
18  	private static final String DEFAULT_CODENAME = "experimental";
19  	/**
20  	 * @deprecated
21  	 * @see #repositoryCodenames
22  	 * @parameter expression="${deb.repository.branch}" default-value="experimental"
23  	 */
24  	protected String repositoryBranch;
25  
26  	/**
27  	 * @parameter expression="${deb.repository.codenames}"
28  	 * @since 1.0.7
29  	 */
30  	protected String[] codenames;
31  
32  	/**
33  	 * @parameter expression="${deb.deploy.skipMissing}" default-value="false"
34  	 * @since 1.0.4
35  	 */
36  	protected boolean skipDeployMissing;
37  
38  	@SuppressWarnings("unchecked")
39  	private void runReprepro() throws IOException, MojoExecutionException
40  	{
41  		File pkgfile = getPackageFile();
42  		if (pkgfile.exists())
43  		{
44  			List<String> codenames = new ArrayList<String>();
45  			if (repositoryBranch != null)
46  			{
47  				codenames.add(repositoryBranch);
48  				getLog().warn("parameter 'repositoryBranch' is now deprecated; use 'codenames' instead");
49  			}
50  			if (this.codenames != null)
51  				codenames.addAll(Arrays.asList(this.codenames));
52  			if (codenames.isEmpty())
53  				codenames.add(DEFAULT_CODENAME);
54  
55  			for (String codename : codenames)
56  				runProcess(new String[]{"reprepro", "--confdir", repreproConfigurationDir.toString(), "--basedir", repository.toString(), "includedeb", codename, pkgfile.toString()}, true);
57  		}
58  		else
59  			getLog().info("Skipping deployment of non-existent package: "+pkgfile);
60  	}
61  
62  	protected void executeDebMojo() throws MojoExecutionException
63  	{
64  		try
65  		{
66  			runReprepro();
67  		}
68  		catch (IOException e)
69  		{
70  			getLog().error(e.toString());
71  			throw new MojoExecutionException(e.getMessage(), e);
72  		}
73  	}
74  }
