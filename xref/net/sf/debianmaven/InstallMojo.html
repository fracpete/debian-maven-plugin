1   package net.sf.debianmaven;
2   
3   import org.apache.maven.plugin.MojoExecutionException;
4   
5   import java.io.IOException;
6   
7   /**
8    * Installs the generated Debian package in the current system.
9    *
10   * Uses external utilities: <a href="http://www.debian.org/doc/manuals/debian-faq/ch-pkgtools.en.html">dpkg</a> and sudo.
11   *
12   * @goal install
13   */
14  public class InstallMojo extends AbstractDebianMojo
15  {
16  	private void runInstall() throws IOException, MojoExecutionException
17  	{
18  		if (getPackageFile().exists())
19  			runProcess(new String[]{"sudo", "dpkg", "--force-confdef", "-i", getPackageFile().toString()}, true);
20  		else
21  			getLog().warn("Debian package file not found; doing nothing.");
22  	}
23  
24  	protected void executeDebMojo() throws MojoExecutionException
25  	{
26  		try
27  		{
28  			runInstall();
29  		}
30  		catch (IOException e)
31  		{
32  			getLog().error(e.toString());
33  			throw new MojoExecutionException(e.getMessage(), e);
34  		}
35  	}
36  }
