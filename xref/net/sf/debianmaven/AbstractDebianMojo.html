17  public abstract class AbstractDebianMojo extends AbstractMojo
18  {
19  	private static final String SKIP_DEB_PROPERTY = "skipDeb";
20  	private static final String RUN_DEB_PROPERTY = "runDeb";

22  	public static class LogOutput implements StreamingProcessOwner
23  	{
24  		private Log log;
25  
26  		public LogOutput(Log log)
27  		{
28  			this.log = log;
29  		}
30  
31  		public StreamingProcessOutputType getOutputType() {
32  			return StreamingProcessOutputType.BOTH;
33  		}
34  
35  		public void processOutput(String line, boolean stdout) {
36  			if (stdout)
37  				log.info(line);
38  			else
39  				log.error(line);
40  		}
41  	}

84  	 * @parameter expression="${deb.package.snapshotRevFile}"

87  	private File snapshotRevisionFile = null;
88  
89  	/** @parameter default-value="false" */
90  	private boolean skip;
91  
92  	private static final DateFormat datefmt = new SimpleDateFormat("yyyyMMddHHmm");
93  
94  	private String snapshotRevision = null;
95  
96  	protected String processVersion(String version)
97  	{
98  		if (snapshotRevision == null)
99  		{
100 			Date revtime = snapshotRevisionFile != null
101 					? new Date(snapshotRevisionFile.lastModified())
102 					: new Date();
103 
104 			snapshotRevision = "+" + datefmt.format(revtime);
105 		}
106 
107 		return version.replaceAll("-SNAPSHOT", snapshotRevision);
108 	}
109 
110 	protected String[] processVersion(String[] versions)
111 	{
112 		String[] result = new String[versions.length];
113 		for (int i=0 ; i<versions.length ; i++)
114 			result[i] = processVersion(versions[i]);
115 
116 		return result;
117 	}
118 
119 	protected File newFile(File dir, String name)
120 	{
121 		return new File(dir.getAbsolutePath() + File.separator + name);
122 	}
123 
124 	protected String getPackageVersion()
125 	{
126 		return processVersion(packageVersion);
127 	}
128 
129 	protected File getPackageFile()
130 	{
131 		return newFile(targetDir, String.format("%s_%s-%s_all.deb", packageName, getPackageVersion(), packageRevision));
132 	}
133 
134 	protected void runProcess(String[] cmd, boolean throw_on_failure) throws IOException, MojoExecutionException
135 	{
136 		ProcessBuilder builder = new ProcessBuilder();
137 		builder.command(cmd);
138 
139 		getLog().info("Start process: " + Arrays.asList(cmd));
140 
141 		try
142 		{
143 			CollectingProcessOutput output = new CollectingProcessOutput();
144 			output.monitor(builder);
145 			int exitval = output.getExitCode();
146 			if (exitval != 0)
147 			{
148 				getLog().warn("Exit code: " + exitval);
149 				getLog().warn("stderr:\n" + output.getStdErr());
150 				getLog().warn("stdout:\n" + output.getStdOut());
151 
152 				if (throw_on_failure)
153 					throw new MojoExecutionException("Process returned non-zero exit code: " + Arrays.asList(cmd));
154 			}
155 		}
156 		catch (MojoExecutionException e)
157 		{
158 			throw e;
159 		}
160 		catch (Exception e) {
161 		  throw new IOException(e);
162 		}
163 	}
164 
165 	protected abstract void executeDebMojo() throws MojoExecutionException;
166 
167 	public final void execute() throws MojoExecutionException
168 	{
169 		if (System.getProperties().containsKey(RUN_DEB_PROPERTY))
170 		{
171 			getLog().info("debian-maven execution forced (-DrunDeb)");
172 			executeDebMojo();
173 		}
174 		else if (skip)
175 		{
176 			getLog().info("debian-maven execution skipped ('skip')");
177 		}
178 		else if (System.getProperties().containsKey(SKIP_DEB_PROPERTY))
179 		{
180 			getLog().info("debian-maven execution skipped (-DskipDeb)");
181 		}
182 		else if (!System.getProperty("os.name").equals("Linux"))
183 		{
184 			getLog().warn("debian-maven execution skipped (non-linux OS)");
185 		}
186 		else
187 		{
188 			executeDebMojo();
189 		}
190 	}
