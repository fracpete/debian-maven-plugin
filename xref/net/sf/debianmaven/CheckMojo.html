3   import org.apache.maven.plugin.MojoExecutionException;

5   import java.io.IOException;

7   /**
8    * Checks whether the generated package complies to style rules.
9    * <br>
10   * Uses external utility: <a href="http://lintian.debian.org/">lintian</a>.
11   *
12   * @goal check
13   * @phase package
14   */
15  public class CheckMojo extends AbstractDebianMojo
16  {
17  	private void runLintian() throws IOException, MojoExecutionException
18  	{
19  		runProcess(new String[]{"lintian", getPackageFile().toString()}, true);
20  	}
21  
22  	protected void executeDebMojo() throws MojoExecutionException
23  	{
24  		try
25  		{
26  			runLintian();
27  		}
28  		catch (IOException e)
29  		{
30  			getLog().error(e.toString());
31  			throw new MojoExecutionException(e.getMessage(), e);
32  		}
33  	}
34  }
